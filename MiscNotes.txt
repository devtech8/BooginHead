Desert.jpg        1024 x 768
shield_white.jpg  674 x 479   
SkullGirl.png     362 x 295

convert Desert.jpg -composite shield_white.png -composite SkullGirl.png output.jpg

# This is one way to create the text...
convert -size 674x479 xc:transparent \
    -pointsize 30 \
    -font Georgia \
    -fill red \
    -size 220x \
    -gravity center \
    caption:'Precious Baby Snookums' \
    text.png

# This works to create the image...
convert -crop 674x479+165+145 \
    -page +0+0 Desert.jpg \
    -page +165+145 shield_white.png \
    -page +331+236 SkullGirl.png \
    -layers merge +repage output.png 

# Another way to create text...
convert -size 674x479 xc:transparent \
    -pointsize 24 \
    -font Georgia \
    -fill green \
    -gravity center \
    -draw "text 0,0 'My precious baby Snookums'" \
    text.png

convert output.png \
    -gravity center \
    -page +0+0 text.png \
    -layers merge +repage output_with_text.png
